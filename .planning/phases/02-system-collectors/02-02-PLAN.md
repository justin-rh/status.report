"""Collector orchestration. Selects platform implementation.
Phase 2: Windows implementation only. Mac stubs reserved for v2.
collect_all(report) is the single entry point called by main.py (Phase 3 wiring).

from __future__ import annotations
from models import AuditReport


def collect_all(report: AuditReport) -> None:
    """Run all collectors in order. Mutates report in place. Never raises.

    Calls collect_hardware first (OS, CPU, RAM, disk, current user),
    then collect_profiles (local user profiles from registry).
    Both functions degrade gracefully — collection_errors accumulates failures.
    """
    from collectors.windows.hardware import collect_hardware, collect_profiles
    collect_hardware(report)
    collect_profiles(report)
```

"""Unit tests for collect_hardware.
WMI and psutil error paths tested via unittest.mock.patch.
psutil happy path runs against real machine (no mock needed — always works).
"""
import pytest
from unittest.mock import patch, MagicMock, PropertyMock
from models import AuditReport
from parsers.name_parser import parse_hostname


def make_report() -> AuditReport:
    return AuditReport(hostname="TEST-PC", parsed_hostname=parse_hostname("TEST-PC"))


class TestCollectOsFields:
    def test_os_version_populated(self):
        from collectors.windows.hardware import collect_hardware
        report = make_report()
        collect_hardware(report)
        assert report.os_version is not None
        assert isinstance(report.os_version, str)

    def test_os_build_populated(self):
        from collectors.windows.hardware import collect_hardware
        report = make_report()
        collect_hardware(report)
        assert report.os_build is not None
        assert "." in report.os_build  # "10.0.19045" format


class TestCollectCpuModel:
    def test_cpu_model_populated_when_wmi_works(self):
        from collectors.windows.hardware import collect_hardware
        mock_proc = MagicMock()
        mock_proc.Name = "  Intel(R) Core(TM) i7-10750H CPU @ 2.60GHz  "
        mock_wmi_instance = MagicMock()
        mock_wmi_instance.Win32_Processor.return_value = [mock_proc]

        # Adapt patch target based on how hardware.py imports wmi (read the file first)
        with patch("collectors.windows.hardware._wmi_module") as mock_wmi_mod:
            mock_wmi_mod.WMI.return_value = mock_wmi_instance
            report = make_report()
            collect_hardware(report)

        assert report.cpu_model == "Intel(R) Core(TM) i7-10750H CPU @ 2.60GHz"
        assert report.collection_errors == []

    def test_cpu_model_none_when_wmi_fails(self):
        from collectors.windows.hardware import collect_hardware
        with patch("collectors.windows.hardware._wmi_module") as mock_wmi_mod:
            mock_wmi_mod.WMI.side_effect = Exception("COM unavailable")
            report = make_report()
            collect_hardware(report)

        assert report.cpu_model is None
        assert len(report.collection_errors) == 1
        error_lower = report.collection_errors[0].lower()
        assert any(kw in error_lower for kw in ("cpu", "wmi", "processor"))


class TestCollectMemoryAndDisk:
    def test_ram_gb_positive(self):
        from collectors.windows.hardware import collect_hardware
        with patch("collectors.windows.hardware._wmi_module"):
            report = make_report()
            collect_hardware(report)
        assert report.ram_gb is not None
        assert report.ram_gb > 0

    def test_disk_fields_none_on_error(self):
        from collectors.windows.hardware import collect_hardware
        with patch("collectors.windows.hardware._wmi_module"):
            with patch("collectors.windows.hardware.psutil.disk_usage",
                       side_effect=PermissionError("access denied")):
                report = make_report()
                collect_hardware(report)

        assert report.disk_total_gb is None
        assert report.disk_free_gb is None
        assert any("disk" in e.lower() for e in report.collection_errors)
```

winreg is mocked — no real registry reads performed.

import sys

from unittest.mock import patch, MagicMock, call
from models import AuditReport
from parsers.name_parser import parse_hostname

# Skip entire module on non-Windows (winreg is Windows-only stdlib)
pytestmark = pytest.mark.skipif(
    sys.platform != "win32",
    reason="winreg only available on Windows"
)


def make_report() -> AuditReport:
    return AuditReport(hostname="TEST-PC", parsed_hostname=parse_hostname("TEST-PC"))


class TestCollectProfiles:
    def test_system_sids_filtered(self):
        """S-1-5-18 (SYSTEM) must not appear in local_profiles; S-1-5-21-* must appear."""
        from collectors.windows.hardware import collect_profiles

        sids_and_paths = {
            'S-1-5-18': r'C:\Windows\system32\config\systemprofile',
            'S-1-5-21-111-222-333-1001': r'C:\Users\john.doe',
            'S-1-5-21-111-222-333-1002': r'C:\Users\jane.smith',
        }
        sids = list(sids_and_paths.keys())

        def mock_enum_key(key, i):
            if i < len(sids):
                return sids[i]
            raise OSError("no more keys")

        def mock_open_key(hive, subkey):
            cm = MagicMock()
            cm.__enter__ = MagicMock(return_value=cm)
            cm.__exit__ = MagicMock(return_value=False)
            return cm

        def mock_query_value(key, name):
            # Return path for the matching SID — key context unavailable so
            # use a workaround: patch at _enumerate_profiles level instead
            pass

        # Patch _enumerate_profiles directly for clean SID-filter test
        def fake_enumerate():
            result = []
            for sid, path in sids_and_paths.items():
                from collectors.windows.hardware import SYSTEM_SIDS
                if sid in SYSTEM_SIDS:
                    continue
                username = path.rstrip('\\').split('\\')[-1]
                if username:
                    result.append(username)
            return result

        with patch("collectors.windows.hardware._enumerate_profiles", side_effect=fake_enumerate):
            report = make_report()
            collect_profiles(report)

        assert 'john.doe' in report.local_profiles
        assert 'jane.smith' in report.local_profiles
        assert 'systemprofile' not in report.local_profiles
        assert report.collection_errors == []

    def test_registry_failure_logs_error(self):
        """OSError opening ProfileList key -> local_profiles=[], error logged."""
        from collectors.windows.hardware import collect_profiles

        with patch("collectors.windows.hardware._enumerate_profiles",
                   side_effect=OSError("Access is denied")):
            report = make_report()
            collect_profiles(report)

        assert report.local_profiles == []
        assert len(report.collection_errors) == 1
        assert "profile" in report.collection_errors[0].lower()

    def test_empty_profile_list_no_error(self):
        """No SIDs in ProfileList -> local_profiles=[], no errors."""
        from collectors.windows.hardware import collect_profiles

        with patch("collectors.windows.hardware._enumerate_profiles", return_value=[]):
            report = make_report()
            collect_profiles(report)

        assert report.local_profiles == []
        assert report.collection_errors == []
