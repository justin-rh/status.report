- disk_free_gb=1.5,  disk_total_gb=10.0 -> severity='WARN', code='DISK_SPACE'  [exactly 15%]

- disk_free_gb=10.0, disk_total_gb=10.0 -> severity='OK',   code='DISK_SPACE'  [100% free]

- disk_free_gb=1.0,  disk_total_gb=None -> severity='OK',   code='DISK_SPACE', detail is not None  [partial None]
